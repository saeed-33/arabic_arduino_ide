متغير ليد : صحيح = 9؛           
متغير سطوع : صحيح = 0؛         
متغير مقدار_التلاشي : صحيح = 5؛ 

دالة إعداد() : فارغ {
  وضع_الطرف(ليد ، مخرج)؛
}

دالة تحديث_السطوع(القيمة_الحالية : صحيح ، التغير : صحيح) : صحيح {
  ارجع القيمة_الحالية + التغير؛
}

دالة حلقة() : فارغ {

  كتابة_تناظرية(ليد ، سطوع)؛

  سطوع = تحديث_السطوع(سطوع،  مقدار_التلاشي)؛

  إذا (سطوع <= 0 أو سطوع >= 255) {
    مقدار_التلاشي = -مقدار_التلاشي؛
  }
  
  تأخير(30)؛
}
